        <button class="btn" onclick="chooseMoney(30000)">30000块</button>
        <div id="result"></div>

    <script>
        function chooseMoney(amount) {
            const resultElement = document.getElementById('result');
            if (amount === 3000) {
                resultElement.textContent = '飞司你挺敢想';
                resultElement.style.color = '#ed8936'; // 橙色
            } else if (amount === 30000) {
                resultElement.textContent = '飞司你去买彩票';
                resultElement.style.color = '#48bb78'; // 绿色
            }
            // 禁用按钮，防止重复点击
            const buttons = document.querySelectorAll('.btn');
            buttons.forEach(btn => btn.disabled = true);
        }
    </script>
